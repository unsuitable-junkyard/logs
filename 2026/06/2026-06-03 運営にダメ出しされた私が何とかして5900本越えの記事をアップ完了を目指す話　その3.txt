import json
from pathlib import Path

log_path = Path(r"C:\blog-tools\repo_uploaded_log.json")

    log = json.load(f)

# URLが空かエンコードされてる5本を修正

fixes = {
    "2020-12-27 多重世界　輪廻で見たこと.txt": f"{base}/2020/12/2020-12-27 多重世界　輪廻で見たこと.txt",
    "2024-10-06 今回は神について考えたいと思います copilot.txt": f"{base}/2024/10/2024-10-06 今回は神について考えたいと思います copilot.txt",

    "2024-10-13 AIとAIのできることについて語らう-20241013-224832.txt": f"{base}/2024/10/2024-10-13 AIとAIのできることについて語らう-20241013-224832.txt",
}

for filename, url in fixes.items():
    if filename in log:
        log[filename]["url"] = url
